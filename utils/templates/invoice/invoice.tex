% The \twocolumnlayout command creates a two-column layout with the specified content and takes three arguments:
% 1. An optional argument that specifies the width of the columns. If this argument is not provided, the width defaults to 1cm.
% 2. The content for the first column.
% 3. The content for the second column.
% The command returns a tabular environment with two columns. The columns are top-aligned and have no extra space on the sides.
\newcommand*{\twocolumnlayout}[3][10cm]{
    \begin{tabular}[h]{@{}p{#1}p{#1}@{}}
        #2 & #3 
    \end{tabular}
}

\newenvironment{infoSection}{\tabular[t]{@{} r @{\hspace{0.5em}} l}}{\endtabular}

% The \infoBox command creates a table row with two columns and takes three arguments:
% 1. An optional argument that specifies the vertical space after the row. If this argument is not provided, the space defaults to 2ex.
% 2. The content for the first column.
% 3. The content for the second column, which is placed inside a minipage. The width of the minipage is 35% of the text width.
% The command returns a table row with the specified content and vertical space.
\newcommand*{\infoBox}[3][2ex]{
    #2 \hspace{3em} #3 \\
    \noalign{\addvspace{#1}}
}
